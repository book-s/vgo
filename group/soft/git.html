# Git

# Git配置 常用命令
git config -l # 配置
git config --system --list # Git系统配置
git config --global --list # Git本地全局配置

# .Gitconfig文件在Linux或MacOSX系统的用户目录下
# Git本地必须配置信息如下
git config --global user.name "name" # 用户名称
git config --global user.email "xxx@xxx.com" # 用户邮箱

Git基本理论（核心）

-- Git 本地有三个工作区域和一个远程仓库
-- 工作目录(Working Directory)
-- 暂存区(Stage Index)
-- 资源区(Repository或Git Directory)
-- 远程Git仓库(Remote Directory)


git status                # 查看所有文件状态
git init                  # 初始化
git add .                 # 添加所有屋内按到暂存区
git add -A                # 添加git仓库暂存区
git commit -m "消息内容"   # 提交暂存区内容到本地仓库 -m 提交信息
git checkout              # 撤销提交
git reset                 # 
git push                  # 提交到本地仓库或远程仓库
git pull                  # 拉取本地仓库最新版本

.gitignore                # 忽略不需要的上传的文件 
*.txt                     # 忽略所有带txt文件内容
node_modules/             # 忽略文件/目录内所有内容
.DS_Store                 # 忽略文件内容
/abc                      # 忽略/文件前目录所有内容
